### 2.2 代码规范

```python
# ✅ 推荐：清晰的命名和日志
import logging

logger = logging.getLogger(__name__)

def process_data(items: list[dict]) -> list[dict]:
    """处理数据项，过滤无效记录。

    Args:
        items: 原始数据列表

    Returns:
        过滤后的有效数据
    """
    logger.info(f"[DataProcessor] 开始处理 {len(items)} 条数据")

    valid_items = [item for item in items if item.get("status") == "active"]

    logger.info(
        f"[DataProcessor] 处理完成: "
        f"输入={len(items)}, 有效={len(valid_items)}, "
        f"过滤={len(items) - len(valid_items)}"
    )
    return valid_items

# ❌ 避免：不清晰的命名和缺少日志
def proc(d):
    return [x for x in d if x.get("status") == "active"]
```
